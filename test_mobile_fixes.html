Carousel Boundary Control:

// Enhanced scroll boundary control
function handleScroll() {
  const cardWidth = cardElement.offsetWidth + 16;
  const maxScrollLeft = (totalCards - 1) * cardWidth;
  
  // Prevent over-scroll past the last card
  if (carousel.scrollLeft > maxScrollLeft) {
    carousel.scrollTo({
      left: maxScrollLeft,
      behavior: 'auto'
    });
  }
  
  // Prevent over-scroll before the first card
  if (carousel.scrollLeft < 0) {
    carousel.scrollTo({
      left: 0,
      behavior: 'auto'
    });
  }
}
